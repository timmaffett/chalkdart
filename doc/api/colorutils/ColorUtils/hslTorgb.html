static List<int> hslTorgb(num hue, num sat, num light) {
  List<double> rgb1 = hslToRgb1Scale(hue, sat, light);
  return [
    (rgb1[0] * 255).round(),
    (rgb1[1] * 255).round(),
    (rgb1[2] * 255).round()
  ];
}
